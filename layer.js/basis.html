1. 无按钮消息弹框。
    layer.msg('hello world');
  2. 有按钮消息弹框。
    layer.alert('hello world');
  3. 无按钮图标弹框。
    layer.msg('hello world', {icon:1});//1-绿勾，2-红叉，3-黄问号，4-灰锁，5-红垮脸，6-绿笑脸，default-黄叹号
  4. 无按钮图标限时弹框。
    layer.msg('hello world',{icon:5, time:500});
  5. 有按钮图标弹框。
    layer.alert('内容', {
    icon: 3, //1-绿勾，2-红叉，3-黄问号，4-灰锁，5-红垮脸，6-绿笑脸，default-黄叹号
    skin: 'layer-ext-moon' //该皮肤由layer.seaning.com友情扩展
});
  6. 自定义按钮图标弹框。
    layer.msg('hello world', {icon: 6, btn: ['Button1', 'Button2', 'Button3']}, function () {
    layer.msg('Response');
});
  7. 确认框。
    layer.confirm('How are you?', {btn:['Good', 'Not good', 'Cancel']},
    function () {layer.msg('Good ?')},
    function () {layer.msg('Not good ?')}
);
