    $("body").delegate("", "keydown", function () {
    if ((event.key == 'd' || event.key == 'Enter') && event.ctrlKey) { //二键
        alert(1);
        event.preventDefault();
    }
    if (event.key == 'e' && event.ctrlKey && event.altKey) { //三键
        alert(2);
        event.preventDefault();
    }
});
