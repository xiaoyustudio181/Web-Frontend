    document.getElementById('parag1')
    document.getElementsByClassName('parags')[0]
    document.getElementsByTagName('p')[0]

    A paragraph.

  2. 获取或设置 DOM 对象的内容。
    parag1.innerHTML //内部内容
    parag1.innerHTML = '<em>Another paragraph.</em>'
    parag1.outerHTML //包括自身的内容
    parag1.outerHTML = '<span>A line.</span>'
  3. 获取或设置 DOM 对象属性值。
    var parag1 = document.getElementById('parag1')
    parag1.id //ID
    parag1.className //类名
    parag1.getAttribute('title') //获取属性值
    parag1.getAttribute('attr1') //获取属性值
    parag1.setAttribute('attr1','hello world') //设置属性值
    parag1.removeAttribute('attr1') //移除属性
  4. 获取 DOM 对象的关联结点。
    parag1.parentNode //父结点
    parag1.childNodes //子结点
    parag1.childNodes[0].childNodes[0].data //子结点的文本结点的内容
    parag1.childNodes[0].childNodes[0].nodeName //子结点的文本结点的结点名
    parag1.attributes //属性结点
    parag1.nodeName //结点名
  5. 获取或设置 DOM 对象样式。
    parag1.style //样式
    parag1.style.color = 'red' //设置样式
    parag1.style.color //获取样式
    parag1.style.fontFamily = 'Comic Sans MS';
    parag1.style.fontSize = '15px';
    parag1.style.backgroundColor = 'pink';
    parag1.style = 'font-family:"Consolas"; font-size:20px; background-color:lightgray;';
  6. 控制表单控件。

    var txt1 = document.getElementById('txt1')
    txt1.value
    txt1.value = '你好世界'
  7. 元素操作。

    var parag2 = document.getElementById('parag2')
    var em1 = document.createElement('em') //新增子结点
    var string1 = document.createTextNode('hello earth') //新增文本结点
    var string2 = document.createTextNode('hi UFO')
    em1.appendChild(string1) //插入子结点
    parag2.appendChild(em1)
    em1.firstChild === string1
    em1.lastChild === string1
    em1.insertBefore(string2,string1) //在指定的子节点前面插入新的子节点
    em1.replaceChild(string2,string1) //替换子结点
    em1.removeChild(string2) //移除子结点
