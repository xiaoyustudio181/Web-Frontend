  2. 普通 JavaScript 对象，JSON(JavaScript Object Notation，JavaScript对象表示法)数据。
    var person = { //定义对象
    name: 'peter',
    age: 24,
    gender: true,
    籍贯: '成都',
    100: '数字键名'
};
    person
    person.name
    person['name']
    person['hobby'] = 'singing'
    Object.keys(person) //显示所有键名
    for (var key in person) { //遍历对象
    var each = person[key];
    console.log(key, '-', each);
}
    delete person.age //删除键
    delete person['100'] //删除键
