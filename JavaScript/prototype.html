1. 定义类。
    function Person(name,age) { //构造函数
    this.name = name;
    this.age = age;
}
Person.prototype.eat = function () { //类方法
    console.log(this.name + ' is eating.');
};
    Person
    typeof Person //function
    Person.prototype
    typeof Person.prototype //object
    var person1 = new Person('Tom', 11); //实例化对象
    person1
    person1.eat()
  2. 类的继承。
    function Student(name, age, school) {
    Person.call(this, name, age); //调用父类构造函数
    this.school = 'no.1 primary school';
}
Student.prototype = new Person(); //继承
Student.prototype.eat = function () {
    Person.prototype.eat.call(this);//调用父类方法
    console.log('... in ' + this.school);
};
    Student
    typeof Student //function
    Student.prototype
    typeof Student.prototype //object
    var student1 = new Student('Jim', 10); //实例化对象
    student1.eat();
    student1 instanceof Student //true
    student1 instanceof Person //true
    student1 instanceof Object //true
    student1 instanceof Function //false
