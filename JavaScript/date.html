  2. 定义时间。new Date(Year, Month, date, Hour, Minute, Second, Millisecond)
    new Date() //现在时间
    new Date(1993, 3, 19, 10, 30, 15, 100) //自定义时间
  3. 获取时间值。
    var d1 = new Date(1993, 3, 19, 10, 30, 15, 100)
    d1.getFullYear() //年
    d1.getMonth()+1 //月
    d1.getDate() //日
    d1.getHours() //时
    d1.getMinutes() //分
    d1.getSeconds() //秒
    d1.getMilliseconds() //毫秒
    d1.getTime() //时间戳
