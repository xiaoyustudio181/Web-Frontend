1. 查看字符串的所有方法。
    console.dir(String)
  2. 获取类型。typeof
    var str = 'abc';
    typeof str
  3. 获取字符串的长度。length
    str = 'abc';
    str.length
  4. 字符串连接(+)。concat
    str = 'abc';
    str + '-de'
    str.concat('-de')
    str.concat('-de', '-fg')
  5. 返回指定索引处的字符。charAt
    str = 'abc';
    str.charAt(1)
  6. 返回指定子串首次出现的索引。indexOf
    str = 'abc';
    str.indexOf('b');
  7. 返回指定子串最后一次出现的索引。indexOf
    str = 'abcb';
    str.lastIndexOf('b');
  8. 检查字符串是否以指定子串开始。startsWith
    str = 'abcb';
    str.startsWith('ab');
  9. 从开头处插入字符串到指定长度。padStart
    str = 'abc';
    str.padStart(10, 'p');
  10. 从结尾处插入字符串到指定长度。str.padEnd
    str = 'abc';
    str.padEnd(10, 'p');
  11. 将字符串转为大写字母。toUpperCase
    str = 'abc';
    str.toUpperCase();
  12. 将字符串转为小写字母。toLowerCase
    str = 'ABC';
    str.toLowerCase();
  13. 重复连接自身指定次数。repeat
    str = 'abc';
    str.repeat(2);
  14. 替换指定子串为新字符串。replace
    str = 'abc';
    str.replace('c', '-ufo');
  15. 以指定字符串分隔成数组。str.split
    str = 'a,b,c';
    str.split(',');
    str.split(',', 2);
  16. 返回截取的字符串。slice [start,end)
    str='abc';
    str.slice(1);
    str.slice(1,2);
    str.slice(-1);
    str.slice(-2,-1);
  17. 返回截取的字符串。substr(start,[length])
    str='abc';
    str.substr(1);
    str.substr(1,2);
    str.substr(-1);
  18. 返回截取的字符串（不支持负数索引）。substring(start,[end])
    str='abc';
    str.substring(1);
    str.substring(1,2);
  19. 返回与正则表达式查找内容匹配的第一个子字符串的位置。search(regex)
    str='abc';
    str.search(/c/);
  20. 返回正则表达式与字符串匹配的信息，没有返回null。match(regex)
    str='abcde';
    str.match(/de$/);
  21. 移除字符串左右的空白符。trim
    str=' abcde ';
    str.length
    str.trim().length;
    str.trimRight().length;
    str.trimLeft().length;
  22. 将字符串作为 URL 组件进行编码。encodeURIComponent
    var str = 'http://www.baidu.com/';
    encodeURIComponent(str)
  23. 将字符串作为 URL 组件进行解码。decodeURIComponent
    var url = 'http%3A%2F%2Fwww.baidu.com%2F';
    decodeURIComponent(url);
