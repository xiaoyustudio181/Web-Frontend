    typeof 12.3 //number
    typeof 'abc' //string
    typeof true //boolean
    typeof [12,'ab'] //object
    typeof {a:12,b:'ab'} //object
    typeof function (){} //function
    typeof null //object //null 显示为 Object 类型而非 null 类型是为了向下兼容
    typeof a //undefined

    0 === false //false
    '' == false //true
    '' === false //false
    [] == false //true
    [] === false //false

    (-1) == false //false
