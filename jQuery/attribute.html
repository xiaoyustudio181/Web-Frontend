• .attr(), removeAttr() 获取、设置、删除元素属性值。
    $('#app').attr('id')
    $('#app').attr('class')
    $('#app').attr('prop1')
    $('#app').attr('prop1',666)
    $('#app').removeAttr('prop1')
  • .prop(), removeProp() 获取、设置、删除元素属性值。它所做的改变不基于HTML标签。
    $('#app2').prop('id')
    $('#app2').prop('class')
    $('#app2').prop('class','default purple')
    $('#app2').prop('prop1')
    $('#app2').prop('prop1',666)
    $('#app2').removeProp('prop1')
  • .prop(), removeProp() 控制表单控件。
    $('#txt1').prop('disabled')
    $('#txt1').prop('disabled',true)
    $('#txt1').prop('disabled',false)
    $('#txt1').prop('checked')
    $('#ch1').prop('checked',true)
    $('#ch1').prop('checked',false)
  • .addClass(), .removeClass(), hasClass(), toggleClass() 给元素增加类。
    $('#app1').attr('class')
    $('#app1').hasClass('red')
    $('#app1').addClass('red')
    $('#app1').toggleClass('red')
    $('#app1').addClass('blue green')
    $('#app1').removeClass('red green')
  • .text(), .html(), .val() 获取或设置值。
    $('#app3').text()
    $('#app3').html()
    $('#app3').text('<strong>hello world</strong>')
    $('#app3').html('<strong>hello world</strong>')
  • .val() 获取或设置值。
    $('#app4').val()
    $('#app4').val('hello world')
